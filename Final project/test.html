module IMP-SYNTAX
    syntax AExp ::=  Int  // builtin
	        |  Id  // builtin
	        |  String  // builtin
	        |  "++"  Id
	        |  Id  "++"
	        >  left:
	        AExp "*" AExp [left]
	        | AExp "/" AExp [left]
	        >  left:
	        AExp "+" AExp [left, strict]
	        | AExp "-" AExp [left]
	        |  "(" AExp ")" [bracket]
   endmodule
 //</syntax>
